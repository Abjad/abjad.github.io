>>> staff = abjad.Staff("c'8 d'8 ~ d'2 e'4")
>>> with abjad.ForbidUpdate(component=staff):
...     for note in staff[:]:
...         pitch_1 = note.written_pitch
...         pitch_2 = pitch_1 + abjad.NamedInterval("M3")
...         pitches = [pitch_1, pitch_2]
...         chord = abjad.Chord(pitches, note.written_duration)
...         abjad.mutate.replace(note, chord)
... 

>>> abjad.wf.wellformed(staff)

class abjad.contextmanagers.RedirectedStreams(stdout=None, stderr=None)[source]Â¶

A context manager for capturing stdout and stderr output.

>>> from io import StringIO
>>> string_io = StringIO()
>>> with abjad.RedirectedStreams(stdout=string_io):
...     print("hello, world!")
... 
>>> result = string_io.getvalue()
>>> string_io.close()
>>> print(result)
hello, world!

Redirected streams context manager is immutable.

class abjad.contextmanagers.Timer(exit_message=None, enter_message=None, print_continuously_from_background=False, verbose=True)[source]Â¶

A timing context manager.

>>> timer = abjad.Timer()
>>> with timer:
...     for _ in range(1000000):
...         x = 1 + 1
... 
>>> timer.elapsed_time  
0.10913205146789551

The timer can also be accessed from within the with block:

>>> with abjad.Timer() as timer:  
...     for _ in range(5):
...         for _ in range(1000000):
...             x = 1 + 1
...         print(timer.elapsed_time)
... 
0.08058810234069824
0.17088818550109863
0.2737002372741699
0.37939023971557617
0.48476219177246094

Timers can be reused between with blocks. They will reset their clock on entering any with block.
