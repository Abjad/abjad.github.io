>>> pcs = [abjad.NamedPitchClass(_) for _ in "c ef g bf d f af".split()]
>>> pitches = abjad.pcollections.voice_vertically(pcs)
>>> staff = abjad.Staff([abjad.Note(_, (1, 8)) for _ in pitches])
>>> abjad.show(staff)
