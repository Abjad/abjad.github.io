    def alphabetize(lists):
        assert isinstance(lists, list)
        for item in lists:
            assert isinstance(item, list), repr(item)

        def key(list_):
            list_ = [_ for _ in list_ if not _.lstrip().startswith("%! ")]
            list_ = [_.removeprefix("%@% ") for _ in list_]
            return list_

        lists_ = lists[:]
        lists_.sort(key=key)
        strings = []
        for list_ in lists_:
            strings.extend(list_)
        return strings

[docs]    def get_contribution_lists(self):
        lists = []
        for site in self:
            for list_ in site:
                if list_:
                    lists.append(list_)
        return lists
